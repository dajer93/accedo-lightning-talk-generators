movieNames = ({
      titleStarts: ['Middle', 'Lilla', 'Apocalipse', 'Silent', 'The woman'],
      titleEnds: ['of the route', 'now', 'of the Earth', 'thousand times'],
      [Symbol.iterator]: function* () {
        for (let titleStart of this.titleStarts) {
          for (let titleEnd of this.titleEnds) yield `${titleStart} ${titleEnd}`;
        }
      }
    });
  
    function printPairs() {
      document.getElementById('displayValue').innerHTML = "";
  
      [...movieNames].forEach(function(movieTile, index) {
        const div = document.createElement('div');
        const span = document.createElement('span');
  
        span.innerHTML = `${movieTile}`;
        div.appendChild(span);
        document.getElementById('displayValue').appendChild(div);
      });
    }
