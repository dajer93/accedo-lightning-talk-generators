possibleParkingSpots = ({
      letters: ['A', 'B', 'C', 'D'],
      [Symbol.iterator]: function* (){
        for (let letter of this.letters) {
          for (let i = 100; i < 102; i++) yield `${letter}${i}`;
          for (let i = 200; i < 205; i++) yield `${letter}${i}`;
          for (let i = 300; i < 305; i++) yield `${letter}${i}`;
        }
      }
    });
  
    function* take(n, iterable) {
      for(let item of iterable){
        if (n <= 0) return;
        n--;
        yield item;
      }
    }
  
    const parkingSpots = [...possibleParkingSpots];
  
    console.log([...take(20, parkingSpots)]);
