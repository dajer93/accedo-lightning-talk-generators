const iteratorExample = {
      count: 0,
      next: function() {
        this.count++;
    
        if (this.count > 5) {
          return { done: true, value: undefined };
        }
    
        return { done: false, value: this.count };
      }
    };
    
    const iterableExample = {
      [Symbol.iterator]: function() {
        return iteratorExample;
      }
    }
    
    const iterableWithGenerator = {
      [Symbol.iterator]: function* () {
        for (i = 1; i <= 5; i++) {
          yield i;
        }
      }
    };
    
    console.log('iterableExample: ', [...iterableExample]);
    console.log('iterableWithGenerator: ', [...iterableWithGenerator]);
